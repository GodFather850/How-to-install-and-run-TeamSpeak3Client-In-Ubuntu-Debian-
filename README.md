```bash
# 1.
cd Downloads

# 2.
chmod +x TeamSpeak3-Client-linux_amd64-3.6.2.run

# 3. Run File
./chmod +x TeamSpeak3-Client-linux_amd64-3.6.2.run
# Prees (q) and Prees y

# 4. Move the created file into the opt executable folder.
mv <File> /opt

# 5. Run Ts3 Client
cd # to exit Downloads Folder
cd /home/$USER/opt
ls -1 # Search To Script  (ts3client_runscript.sh)
bash ts3client_runscript.sh # run ts3

```
